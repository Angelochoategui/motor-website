#### Sidebar Layout. A CSS Sidebar Layout Component

## Usage

A collapsable sidebar component which hides additional content beside a page.

```javascript

import { Sidebar } from @motor-js/core
//...
<Sidebar
  collapsable
  padding="100px 5px"
  isOpen={isOpen}
  backgroundColor="lightblue"
>
  Sidebar Content
</Sidebar>

```

## Example


### Simple sidebar

Render a simple sidebar

```jsx
function SideBarDemo() {
  const { isOpen, toggle } = useSidebar();
  return (
    <Motor config={config}>
      <div style={{ overflow: "hidden", height: "300px", width: "500px" }}>
        <Sidebar
          width="30%"
          collapsable
          direction="column"
          padding="15px 12px"
          isOpen={isOpen}
          backgroundColor="altGray1"
          justifyContent="top"
          border={{ color: "brand" }}
        >
          <Box direction="column" overflow="visible">
              <Button onClick={toggle} size={25}>
                Hide Sidebar
              </Button>
          </Box>
        </Sidebar>
        <Button onClick={toggle}>Show Sidebar</Button>
      </div>
    </Motor>
  );
}
```

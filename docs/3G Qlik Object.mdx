#### Load an object from your Qlik app.

## Usage

```javascript

import { QlikObject } from @motor-js/core

//...

<QlikObject height='300px' width='80%' id='NtQvFrh' />

```

## Props

<table>
  <tr>
    <th>Prop</th>
    <th>Description</th>
    <th>Options / Example</th>
  </tr>
  <tr>
    <td><code>id</code></td>
    <td>id of your Qlik object to be loaded</td>
    <td>string</td>
  </tr>
  <tr>
    <td><code>height</code></td>
    <td>height of the object</td>
    <td>string</td>
  </tr>
  <tr>
    <td><code>width</code></td>
    <td>width of the object</td>
    <td>string</td>
  </tr>
  <tr>
    <td><code>minWidth</code></td>
    <td>minWidth of the object</td>
    <td>string</td>
  </tr>
  <tr>
    <td><code>minHeight</code></td>
    <td>minHeight of the object</td>
    <td>string</td>
  </tr>
  <tr>
    <td><code>border</code></td>
    <td>border</td>
    <td>string</td>
  </tr>
  <tr>
    <td><code>borderRadius</code></td>
    <td>border radius</td>
    <td>string</td>
  </tr>
  <tr>
    <td><code>margin</code></td>
    <td>margin</td>
    <td>string</td>
  </tr>
  <tr>
    <td><code>noSelections</code></td>
    <td>toggle whether you can make selections on the chart</td>
    <td>boolean</td>
  </tr>

    <td><code>noInteraction</code></td>

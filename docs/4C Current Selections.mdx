#### Current Selections Box. Returns the current app selections.

## Usage

```javascript

import { CurrentSelections } from @motor-js/core

//...

<CurrentSelections />

```

## Examples

### Simple Example

```jsx
function SelectionsTest() {
  return (
    <Motor config={config}>
      <CurrentSelections />
    </Motor>
  );
}
```

### Change of Theme

```jsx
function SelectionsTest() {
  return (
    <Motor 
      config={config}
      theme={{
          selections: {
            color: {
              defaultFont: '#0000ff',
              fontTitle: '#ebebff',
              item: '#ebebff',
              clear: 'white',
              border: '#ebebff',
            },
            wrapper: {
              backgroundColor: '#ebebff',
              borderColor: '#ebebff',
            },
            item: {
              backgroundColor: '#0000ff',
            },
          }
      }}
    >
      <Filter 
        dimension={['Product Sub Group Desc']}
        label='Product Group'
      />
      <CurrentSelections size='large'/>
    </Motor>
  );
}
```
